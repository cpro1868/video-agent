- [ ] **Step 2: Add cache TTL (time-to-live) support**

```python
# Add TTL support to cache
def set_cached(key: str, value: Any, ttl: int = 3600) -> None:
    """Set cached value with TTL in seconds."""
    import time
    cache[key] = {"value": value, "expires": time.time() + ttl}

def get_cached(key: str) -> Any | None:
    """Get cached value if not expired."""
    import time
    if key in cache:
        entry = cache[key]
        if time.time() < entry["expires"]:
            return entry["value"]
        del cache[key]
    return None
```
